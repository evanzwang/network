Apparently, there are 2 things that matter:
- The system your crewed science ship ends in must not have been surveyed by you
- Your crewed science ship(s) must end up in 10 different systems in total

Basically, the achievement should be:
"As a Slingshot to the Stars empire, send crewed science ships into 10 different unknown solar systems with Quantum Catapults."

That is if I'm understanding this right:
Code: 
# Into the Unknown Achievement
fleet_event = {
    id = action.430
    hide_window = yes

    is_triggered_only = yes

    trigger = {
        owner = { has_origin = origin_star_slingshot }
        any_controlled_ship = {
            is_ship_class = shipclass_science_ship
        }
        exists = leader
        solar_system = {
            NOT = { has_star_flag = into_the_unknown_flag_@root.owner }
            is_surveyed = {
                who = root.owner
                status = no
            }
        }
    }

    immediate = {
        solar_system = {
            set_star_flag = into_the_unknown_flag_@root.owner
        }
        owner = {
            change_variable = {
                which = into_the_unknown_counter
                value = 1
            }
            if = {
                limit = {
                    check_variable = {
                        which = into_the_unknown_counter
                        value >= 10
                    }
                }
                set_country_flag = into_the_unknown_achievement
            }
        }
    }
}
